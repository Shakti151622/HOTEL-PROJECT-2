Generate room QR codes → guest scans → order hits dashboard live

Flow
1
Print QR and place in room. Guest scans with phone camera.
2
Guest page opens with their room number pre-filled. They browse & order.
3
Order hits Supabase → dashboard shows toast + card instantly via Realtime.
⚠️ Run once in Supabase SQL Editor (if table not created yet)
create table if not exists orders (
  id          text primary key,
  room        text        not null,
  guest       text        not null default 'Guest',
  items       jsonb       not null default '[]',
  total       integer     not null default 0,
  status      text        not null default 'new',
  placed_at   timestamptz not null default now()
);
alter table orders enable row level security;
create policy "guests insert" on orders for insert with check (true);
create policy "staff select" on orders for select using (true);
create policy "staff update" on orders for update using (true);
alter publication supabase_realtime add table orders;
